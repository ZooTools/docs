```ruby Ruby
require 'net/http'
require 'json'

endpoint = URI('https://api.zootools.co/v1/transactional')
token = 'YOUR_BEARER_TOKEN'

payload = {
  name: 'password-reset',
  to: ['jorge@zootools.co'],
  data: {
    passwordResetUrl: 'example.com/reset-password?token=123123123'
  }
}

http = Net::HTTP.new(endpoint.host, endpoint.port)
http.use_ssl = true

request = Net::HTTP::Post.new(endpoint.path, { 'Authorization' => "Bearer #{token}", 'Content-Type' => 'application/json' })
request.body = payload.to_json

response = http.request(request)

if response.code.to_i == 200
  puts 'API response:', response.body
else
  puts 'API request failed:', response.code, response.body
end
```

```python python
import requests

endpoint = 'https://api.zootools.co/v1/transactional'
token = 'YOUR_BEARER_TOKEN'

payload = {
    'name': 'password-reset',
    'to': ['jorge@zootools.co'],
    'data': {
        'passwordResetUrl': 'example.com/reset-password?token=123123123'
    }
}

headers = {
    'Authorization': f'Bearer {token}',
    'Content-Type': 'application/json'
}

response = requests.post(endpoint, json=payload, headers=headers)

if response.status_code == 200:
    print('API response:', response.json())
else:
    print('API request failed:', response.status_code, response.text)
```
